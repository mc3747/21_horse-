1_对象的基本组成

let me = {
        name : 'pingan',
        eat: function(){
            console.log('eat eat eat!!!');
        }
    }
1_属性
    1_获取属性的方法1->点表示法:me.name = "leo";   // me => {name: "leo"}
    2_获取属性的方法1->点括号表示法:me["name"] = "leo";// me => {name: "leo"}
2_方法
    me.eat();

2_属性的操作

1_获取对象的所有属性方法:
    1 for...in 循环(对象及原型链中所有可枚举的属性)
        let obj = {a:1, b:2, c:3};
            for (let i in obj) {
              console.log("obj." + i + " = " + obj[i]);
            }
    2 Object.keys(o) (对象中所有属性名称的数组(不包括原型链))
        let obj = {a:1, b:2, c:3};
        let arr = Object.keys(obj);
    3 Object.getOwnPropertyNames(o):它包含了对象 o 所有拥有的属性（无论是否可枚举）的名称
    let obj = {a:1, b:2, c:3};
    let arr = Object.getOwnPropertyNames(obj);

  2_删除属性:delete,
    继承自父类的属性不能删除,
    Object.defineProperty定义的属性不能删除
    使用: delete obj4.a;

  3_判断属性是否存在
    1.in方法：自己和继承的都包含
        console.log('toString' in obj4);
    2.hasOwnProperty方法：只包括自己的属性
        console.log(obj4.hasOwnProperty('toString'));

3_属性的简洁表示

1.普通对象
    let b = { a };  
    // 等同于let b = { a : a };     
 2.函数中返回对象
    function f(a, b){
         return {a, b}; 
    }
    // 等同于
    function f (a, b){
        return {a:a ,b:b};
    }
  3.对象中的函数属性
    let a = {
            fun () {
                return 'leo';
            }
        }
    // 等同于
    let a = {
            fun : function(){
                return 'leo';
            }
        }

4_对象的浅拷贝和深拷贝

1_深拷贝:值拷贝
2_浅拷贝:对象拷贝(地址拷贝)
3_直接赋值是浅拷贝
    var obj1 = {
    name: 'shen'
    }
    var obj2 = obj1
    obj2.name = 'shenzhiyong'
4_实现对象深拷贝的方法
    1.使用json方法:只适应普通对象,属性中有函数就不行了。
        var obj1 = {
            name: 'shen'
        }
        var obj2 = JSON.parse(JSON.stringify(obj1))
        obj2.name = 'shenzhiyong'
    2.以下只适用没有嵌套的对象
        1_es6解析结构 「...」
            var obj1 = {
                name: 'shen',
                show: function (argument) {
                    console.log(1)
                }
            }
            var obj2 = { ...obj1 }
            obj2.name = 'shenzhiyong'
            console.log('obj1:', obj1)  // obj {name: "szy", show: ƒ}
            console.log('obj2:', obj2)   // obj2 {name: "shenzhiyong",show:ƒ}
        2_Object.assign()
                var obj1 = {
                name: 'shen',
                show: function (argument) {
                    console.log(1)
                }
            }
            var obj2 = Object.assign(obj1)
            obj2.name = 'shenzhiyong'
            console.log('obj1:', obj1)  // obj {name: "szy", show: ƒ}
            console.log('obj2:', obj2)   // obj2 {name: "shenzhiyong",show:ƒ}
        3_数组中的slice() & concat()
            var arr1 = [1,2,3]
            var arr2 = arr1.slice()  // 方法一
            // var arr2 = arr1.concat()  //方法二
            arr2.push(4)
            console.log('arr1:', arr1)  // arr1: [1, 2, 3]
            console.log('arr2:', arr2)  // arr1: [1, 2, 3, 4]

    3.终极解决办法手写递归方法
            // 定义一个深拷贝函数  接收目标target参数
        function deepClone(target) {
            // 定义一个变量
            let result;
            // 如果当前需要深拷贝的是一个对象的话
            if (typeof target === 'object') {
            // 如果是一个数组的话
                if (Array.isArray(target)) {
                    result = []; // 将result赋值为一个数组，并且执行遍历
                    for (let i in target) {
                        // 递归克隆数组中的每一项
                        result.push(deepClone(target[i]))
                    }
                 // 判断如果当前的值是null的话；直接赋值为null
                } else if(target===null) {
                    result = null;
                 // 判断如果当前的值是一个RegExp对象的话，直接赋值    
                } else if(target.constructor===RegExp){
                    result = target;
                }else {
                 // 否则是普通对象，直接for in循环，递归赋值对象的所有值
                    result = {};
                    for (let i in target) {
                        result[i] = deepClone(target[i]);
                    }
                }
             // 如果不是对象的话，就是基本数据类型，那么直接赋值
            } else {
                result = target;
            }
             // 返回最终结果
            return result;
        }

5_判断两个对象是否相等

1, ==:表示值相等
2, ===:表示严格值相等
3, Object.is(a,b):表示a,b是否指向同一个指针对象
     var a = {
       id:1
     };
     var b = a;
     console.log(Object.is(a,b));   //true
 4,Object.is(a,b):不能判断没有拷贝关系的对象
     var a = {
       id:1
     };
     var b = {
       id:1
     }
     console.log(Object.is(a,b));   //false
 5,终极解决办法:
     只要两个对象的名和键值都相同。那么两个对象的内容就相同
     封装通用方法

6_创建对象的方法

1，基本写法:new Object的方式
    var obj1 = new Object();
    obj1.name = 'wendingding';
    obj1.age = 18;

2, Object.create(proto, [propertiesObject]) 的方法
    创建一个新对象，使用现有的对象来提供新创建的对象的 __proto __

2，字面量的方式创建对象
    var obj2 = {name:"wendingding",age:18};

3,字符串转对象:eval函数创建对象:大括号外部包一层小括号
     let demo = eval("({foo:2})")
     console.log(demo.foo)

4,json转对象:JSON.parse
    var obj = JSON.parse('{"name":"小明","age":14}');
    console.log(obj.name)

5，内置构造函数创建对象
    1_有以下类型:
        String
        Number
        Boolean
        Date
        Array
        Function
        Object
        RegExp
    2_示例
        var s1 = 'abc';//s1是字符串
        var obj3 = new String('abc');//obj3是字符串对象

6，封装工厂函数创建对象
    function createBookNew (name,price) {
        var book = new Object();
        book.name = name;
        book.price = price;
        return book;//一定要加上return
    }
    var book3 = createBookNew("悟空传","28.00");

5，定义构造函数创建对象
    //1_构造方法
    function CreateBook (name,price,author,press) {
        this.name = name;
        this.price = price;
        this.author = author;
        this.press = press;
        this.read = function () {
            console.log("我的书名为:"+this.name+",作者为"+this.author+"....");
        };
    }
    //2_创建对象
    var bookObj = new CreateBook("声名狼藉者的的生活","42.00","福柯","北京大学出版社");
    //3_对象调用方法
    bookObj.read();
