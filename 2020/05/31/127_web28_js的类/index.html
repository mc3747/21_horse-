1.class的本质

1_ES6中的class可以看作只是一个语法糖
2_ES6的类的所有方法都是定义在prototype属性上，调用类的实例的方法，其实就是调用原型上的方法,上例中a.constructor === P.prototype.constructor; // true
3_constructor()是类的默认方法，通过new实例化时自动调用执行，一个类必须有constructor()方法，否则一个空的constructor()会默认添加。constructor()方法默认返回实例对象(即this)

2.实现类的私有方法和私有属性

1_使用命名区分:变量名前添加_，但是不保险，外面也可以调用到
    class P {
            // 公有方法
            f1 (x) {
                this._x(x);
            }
            // 私有方法
            _x (x){
                return this.y = x;
            }
        }
2_将私有方法移除模块，再在类内部调用call方法。
    class P {
            f1 (x){
                f2.call(this, x);
            }
        }
        function f2 (x){
            return this.y = x;
        }
3_使用Symbol为私有方法命名
    const a1 = Symbol('a1');
    const a2 = Symbol('a2');
    export default class P{
        // 公有方法
        f1 (x){
            this[a1](x);
        }
        // 私有方法
        [a1](x){
            return this[a2] = x;
        }
    }

3.类的this指向

1,类内部方法的this默认指向类的实例，但单独使用该方法可能报错，因为this指向的问题
2,解决的办法1:在类里面绑定this
    class P {
        constructor(){
            this.name = this.name.bind(this);
        }
    }
3,解决的办法2:使用箭头函数
    class P{
        constructor(){
            this.name = (name = 'leo' )=>{
                this.print(`my name is ${name}`)
            }
        }
    }

4.类的属性

1_ES6中明确规定，Class内部只有静态方法没有静态属性
2_只能通过下面实现
// 正确写法
    class P {}
    P.a1 = 1;
    P.a1;      // 1

    // 无效写法
    class P {
        a1: 2,          // 无效
        static a1 : 2,  // 无效
    }
    P.a1;      // undefined

5.类的方法

1_Class的getter和setter方法
   ->使用get和set关键词对属性设置取值函数和存值函数，拦截属性的存取行为
    class P {
            constructor (){ ... }
            get f (){
                return 'getter';
            }
            set f (val) {
                console.log('setter: ' + val);
            }
        }

        let a = new P();
        a.f = 100;   // setter : 100
        a.f;          // getter

    2_Class的generator方法
        ->只要在方法之前加个(*)即可
        class P {
                constructor (...args){
                    this.args = args;
                }
                *[Symbol.iterator](){
                    for (let arg of this.args){
                        yield arg;
                    }
                }
            }
            for (let k of new P('aa', 'bb')){
                console.log(k);
            }
            // 'aa'
            // 'bb'

        3_Class的静态方法
            ->由于类相当于实例的原型，所有类中定义的方法都会被实例继承，若不想被继承，只要加上static关键字，只能通过类来调用，即“静态方法”。
            class P (){
                    static f1 (){ return 'aaa' };
                }
                P.f1();    // 'aa'
                let a = new P();
                a.f1();    // TypeError: a.f1 is not a function

        4_静态方法中的this
            class P {
                static f1 (){
                    this.f2();
                }
                static f2 (){
                    console.log('aaa');
                }
                f2(){
                    console.log('bbb');
                }
            }
            P.f2();  // 'aaa'

        5_静态方法可以被子类继承，或者super对象中调用
            class P{
                    static f1(){ return 'leo' };
                }
                class Q extends P { ... };
                Q.f1();  // 'leo'

                class R extends P {
                    static f2(){
                        return super.f1() + ',too';
                    }
                }
                R.f2();  // 'leo , too'

6.类的继承

1_基本使用
    1.使用extends调用;
    2.子类必须在constructor()调用super()否则报错，并且只有super方法才能调用父类实例，
    3.父类的静态方法，子类也可以继承到
    class P { 
        constructor(x, y){
            // ...
        }
        f1 (){ ... }
    }
    class Q extends P {
        constructor(a, b, c){
            super(x, y);  // 调用父类 constructor(x, y)
            this.color = color ;
        }
        f2 (){
            return this.color + ' ' + super.f1(); 
            // 调用父类的f1()方法
        }
    }
2_super关键字:既可以当函数使用，还可以当对象使用
    class P {
            f (){ return 2 };
        }
        class R extends P {
            constructor (){
                super();//当函数调用，代表父类的构造函数，但必须执行一次
                console.log(super.f()); // 当对象调用，指向原型对象，在静态方法中指向父类。
            }
        }
        let a = new R()
