117_web18_js的对象

发表于 2020-05-29 | 分类于 前端 | | 阅读次数:
字数统计: 1.1k | 阅读时长 ≈ 4

##1_对象的基本组成
let me = {
name : ‘pingan’,
eat: function(){
console.log(‘eat eat eat!!!’);
}
}
1_属性
1_获取属性的方法1->点表示法:me.name = “leo”; // me => {name: “leo”}
2_获取属性的方法1->点括号表示法:me[“name”] = “leo”;// me => {name: “leo”}
2_方法
me.eat();

##2_属性的操作
1_获取对象的所有属性方法:
1 for…in 循环(对象及原型链中所有可枚举的属性)
let obj = {a:1, b:2, c:3};
for (let i in obj) {
console.log(“obj.” + i + “ = “ + obj[i]);
}
2 Object.keys(o) (对象中所有属性名称的数组(不包括原型链))
let obj = {a:1, b:2, c:3};
let arr = Object.keys(obj);
3 Object.getOwnPropertyNames(o):它包含了对象 o 所有拥有的属性（无论是否可枚举）的名称
let obj = {a:1, b:2, c:3};
let arr = Object.getOwnPropertyNames(obj);

2_删除属性:delete,
  继承自父类的属性不能删除,
  Object.defineProperty定义的属性不能删除
  使用: delete obj4.a;

3_判断属性是否存在
  1.in方法：自己和继承的都包含
      console.log('toString' in obj4);
  2.hasOwnProperty方法：只包括自己的属性
      console.log(obj4.hasOwnProperty('toString'));

##3_属性的简洁表示
1.普通对象
let b = { a };
// 等同于let b = { a : a };
2.函数中返回对象
function f(a, b){
return {a, b};
}
// 等同于
function f (a, b){
return {a:a ,b:b};
}
3.对象中的函数属性
let a = {
fun () {
return ‘leo’;
}
}
// 等同于
let a = {
fun : function(){
return ‘leo’;
}
}

##4_对象的浅拷贝和深拷贝
1_深拷贝:值拷贝
2_浅拷贝:对象拷贝(地址拷贝)
3_直接赋值是浅拷贝
var obj1 = {
name: ‘shen’
}
var obj2 = obj1
obj2.name = ‘shenzhiyong’
4_实现对象深拷贝的方法
1.使用json方法:只适应普通对象,属性中有函数就不行了。
var obj1 = {
name: ‘shen’
}
var obj2 = JSON.parse(JSON.stringify(obj1))
obj2.name = ‘shenzhiyong’
2.以下只适用没有嵌套的对象
1_es6解析结构 「…」
var obj1 = {
name: ‘shen’,
show: function (argument) {
console.log(1)
}
}
var obj2 = { …obj1 }
obj2.name = ‘shenzhiyong’
console.log(‘obj1:’, obj1) // obj {name: “szy”, show: ƒ}
console.log(‘obj2:’, obj2) // obj2 {name: “shenzhiyong”,show:ƒ}
2_Object.assign()
var obj1 = {
name: ‘shen’,
show: function (argument) {
console.log(1)
}
}
var obj2 = Object.assign(obj1)
obj2.name = ‘shenzhiyong’
console.log(‘obj1:’, obj1) // obj {name: “szy”, show: ƒ}
console.log(‘obj2:’, obj2) // obj2 {name: “shenzhiyong”,show:ƒ}
3_数组中的slice() & concat()
var arr1 = [1,2,3]
var arr2 = arr1.slice() // 方法一
// var arr2 = arr1.concat() //方法二
arr2.push(4)
console.log(‘arr1:’, arr1) // arr1: [1, 2, 3]
console.log(‘arr2:’, arr2) // arr1: [1, 2, 3, 4]

3.终极解决办法手写递归方法

4.使用第三方库:jQuery.extend 和 lodash等

##5_判断两个对象是否相等
1, ==:表示值相等
2, ===:表示严格值相等
3, Object.is(a,b):表示a,b是否指向同一个指针对象
var a = {
id:1
};
var b = a;
console.log(Object.is(a,b)); //true
4,Object.is(a,b):不能判断没有拷贝关系的对象
var a = {
id:1
};
var b = {
id:1
}
console.log(Object.is(a,b)); //false
5,终极解决办法:
只要两个对象的名和键值都相同。那么两个对象的内容就相同
封装通用方法

##6_创建对象的方法
1，基本写法:new Object的方式
var obj1 = new Object();
obj1.name = ‘wendingding’;
obj1.age = 18;

2，字面量的方式创建对象
    var obj2 = {name:"wendingding",age:18};

3,字符串转对象:eval函数创建对象:大括号外部包一层小括号
     let demo = eval("({foo:2})")
    console.log(demo.foo)

4,json转对象:JSON.parse
    var obj = JSON.parse('{"name":"小明","age":14}');
    console.log(obj.name)

5，内置构造函数创建对象
    1_有以下类型:
        String
        Number
        Boolean
        Date
        Array
        Function
        Object
        RegExp
    2_示例
        var s1 = 'abc';//s1是字符串
        var obj3 = new String('abc');//obj3是字符串对象

6，封装工厂函数创建对象
    function createBookNew (name,price) {
        var book = new Object();
        book.name = name;
        book.price = price;
        return book;//一定要加上return
    }
    var book3 = createBookNew("悟空传","28.00");

5，定义构造函数创建对象
    //1_构造方法
    function CreateBook (name,price,author,press) {
        this.name = name;
        this.price = price;
        this.author = author;
        this.press = press;
        this.read = function () {
            console.log("我的书名为:"+this.name+",作者为"+this.author+"....");
        };
    }
    //2_创建对象
    var bookObj = new CreateBook("声名狼藉者的的生活","42.00","福柯","北京大学出版社");
    //3_对象调用方法
    bookObj.read();

##1_常量和变量
1,let,var,const
ES6及以后let和var变量作用域不一样
2,变量声明提升:
将变量的声明提升到函数或语句的顶部，并返回undefined直到变量被初始化操作。
ES5及以前采用,ES6及以后没有
3,函数声明提升
函数声明有3种方式
1,函数声明
f(); // ‘hi leo’
function f(){
console.log(‘hi leo’);
};
2,函数表达式声明
f(); // 报错
var f = function(){
console.log(‘hi leo’);
};
3,函数对象声明
f(); // 报错
var f = new Function(){
console.log(‘hi leo’);
};

##2_数据类型
1_六种原型数据类型：
1.Boolean : 布尔值，true和false;
2.null : 对大小写敏感（即null/Null/NULL完全不同）;
3.undefined : 空类型，变量未定义时候的值；
4.Number : 数值类型，如100;
5.String : 字符串类型，如’hi pingan’;
6.Symbol（ES6新增） : 表示一种唯一且不可变的数据;
2_引用数据类型
1.狭义的对象Object
2.数组
3.函数

##3_字面量
1.数组字面量:let a = [‘hi’,’leo’];
2.布尔字面量:let a = true;
3.浮点数字面量:let a = 0.5(正常表示;带e指数表示)
4.整数字面量:let a = 5(不同进制表示)
5.对象字面量:let a ={1:8} (键值对表)
1_点语法取值
2_中括号取值
6.正则字面量:let a = /ab+c/(两个斜杠)
7.字符串字面量
1_单引号:’aa’
2_双引号:”aa”
3_模板自变量hello${a}

##4_函数

###4.1 定义
1，函数声明
function f(){
console.log(‘hi leo’);
};
2，函数表达式(函数字面量)
var f = function(){
console.log(‘hi leo’);
};
3，使用Function构造函数创建
var f = new Function(){
console.log(‘hi leo’);
};

###4.2 调用方式
1, 普通函数调用模式
f(5,6)
2, 对象方法调用模式
var o1 = {
name: “文顶顶”,
showName: function() {
console.log(‘对象方法调用：’ +this.name);
}
};
o1.showName();
3, 构造函数调用模式
function Person() {
console.log(‘构造函数调用：’ +this);
}
new Person();
4, 上下文的调用模式
//函数定义
function f4(a, b) {
console.log(a, b, a + b);
console.log(this); //使用上下文模式调用时，this被绑定给o对象
console.log(this.name); //wendingding
}
//创建对象
var o2 = {
name: “wendingding”
};
//函数上下文调用
f4.apply(o2, [1, 2]);//f4中的this指向o2
f4.call(o2, 3, 4);//f4中的this指向o2
f4(1,2);//f4中的this指向全局的window

###4.3 默认属性
1，this
2，arguments：参数数组
3，length参数个数
4，name函数名称

###4.4 默认方法
1，toString：返回函数整体字符串
2，apply
3，call

###4.5 特殊参数
1_默认参数
function f(a, b = 1){
return a * b;
}
2_不确定参数
function f (a, …b){
console.log(a, b);
}
f(1,2,3,4); // a => 1 ,b => [2, 3, 4]

###4.6 箭头函数
1_写法
// 1_没参数
let f = () => 1;
等同于let f = function (){return 1};
// 2_有1个参数:
let f = v => v;
等同于let f = function (v){return v};
// 3_有多个参数
let f = (v, i) => {return v + i};
等同于let f = function (v, i){return v + i};
2_注意点
1_箭头函数内的this总是指向定义时所在的对象，而不是调用时。
2_箭头函数不能当做构造函数，即不能用new命令，否则报错。
3_箭头函数不存在arguments对象，即不能使用，可以使用rest参数代替。
3_箭头函数不能使用yield命令，即不能用作Generator函数。
