1.数组的创建

1_使用Array关键字(可以省略new关键字)
   1.1 new Array();
        let arr = new Array(); 
        arr[0] = 'a'
   1.2 new Array(size);
        let arr = new Array(2); 
        arr[0] = 'a'
   1.3 new Array(element0, element1, ..., elementn);
        let arr = new Array('2','6'); 
2_使用字面量
    2.1 定义再赋值
        let arr = []; 
        arr[0] = '1' 
        console.log(arr);  
    2.2 定义的时候赋值
        let arr = ['q','w']; 
3_Array.from()
    将 类数组对象 和 可遍历的对象，转换成真正的数组
        // 1_类数组对象
        let a = {
            '0':'a',
            '1':'b',
            length:2
        }
        let arr = Array.from(a); //[ 'a', 'b' ]

        // 2_可遍历的对象
        let a = Array.from([1,2,3]); //[ 1, 2, 3 ]
        let b = Array.from({length: 3}); //[ undefined, undefined, undefined ]
        let c = Array.from([1,2,3]).map(x => x * x); //[ 1, 4, 9 ]
        let d = Array.from([1,2,3].map(x => x * x)); //[ 1, 4, 9 ]
    4_Array.of()
        将一组数值，转换成数组;弥补Array方法参数不同导致的差异
        let arr = new Array(2); //1个参数时，为指定数组长度,结果为[,]
        let arr = Array.of(2);结果为[2]
        let arr = Array.of(1,2,3);结果为[1,2,3]
    5_fill()
        第二个和第三个参数指定填充的起始位置和结束位置
       let a = new Array(3).fill('a');   // ['a','a','a']
       [1,2,3].fill('a',1,2);//  [1, "a", 3]

2.数组的默认属性

1_constructor    返回对创建此对象的数组函数的引用。
2_length    设置或返回数组中元素的数目。
    可通过arr.length = 0来清空数组
3_prototype    使您有能力向对象添加属性和方法。

3.数组的遍历

1_普通for循环
    for(j = 0; j < arr.length; j++) {} 
2_优化版for循环
    for(j = 0,len=arr.length; j < len; j++) {}
3_for...in循环
    for(j in arr) {}
4_for...of循环
    for(let value of arr) {  });
5_for...each
    1_arr.forEach(callback) 接收一个回调方法。
        callback(val, index, array) : 接收三个参数：
        val : 当前处理的元素；
        index : 当前处理的元素的索引；
        array : 正在处理的数组；
    2_示例
        let arr = ["pingan", "leo", "robin"];
        arr.forEach(function(val, i, array){
                console.log(`第${i}个元素是：${val}；`)
                console.log(arr)
        })
6_map循环
        let arr = ["pingan", "leo", "robin"];
        arr.map(function(val, i, array){
                console.log(`第${i}个元素是：${val}；`)
                console.log(arr)
        })
7_entries(),keys(),values()遍历

        console.log(e)
    }

4,数组合并

语法:arr.concat(a1, a2, ..., an);
参数:
    arr：目标数组；
    a1,a2,...,an：需要合并的元素；

5,数组元素之间拼接成一个字符串

语法: arr.join(sep);
参数:
    arr：目标数组；
    sep：连接的分隔符，默认值为“,”；
示例:
    let arr = ["pingan", "leo", "robin"];
    arr.join(","); // "pingan,leo,robin"

6,颠倒数组中元素的顺序，改变原数组

语法:arr.reverse()
示例:
    let arr = [1, 2, 3, 4];
    arr.reverse();  // [4, 3, 2, 1]

7,对数组排序

1_sort()
    如果调用该方法时没有使用参数，将按字母顺序对数组中的元素进行排序
    比较函数应该具有两个参数 a 和 b，其返回值如下：
        若 a 小于 b，在排序后的数组中 a 应该出现在 b 之前，则返回一个小于 0 的值。
        若 a 等于 b，则返回 0。
        若 a 大于 b，则返回一个大于 0 的值   
2_示例
    let a1 = [1,3,6,9,10];
    a1.sort(); // [1, 10, 3, 6, 9]
    a1.sort(function(a,b){
        return a > b ? 1 : a < b ? -1 : 0;
    })     // [1, 3, 6, 9, 10]
3_注意:不破坏原数组的结构

8,通过索引来切割数组:两个索引之间(slice)

1_语法:slice(start,end)
    start: 必需，指定起始索引，若负数则从数组最后开始算起，-1为倒数第一位，-2为倒数第二位，以此类推。
    end： 可选，指定结束索引，若没传则表示到数组结束
2_示例:
    let arr = [1, 2, 3, 5, 6];
    let a2 = arr.slice(2,3);  // [3]
    console.log(arr) //
3_注意:不破坏原数组的结构

9,通过索引来切割数组:起始索引+数量(splice)

1_语法:
     arr.splice(index, num, a1, a2,...,an);
    参数：
        index: 必需，起始位置的索引，若负数则从数组最后开始算起；
        num：必需，删除的数量，若为0则不删除；
        a1,a2,...an：可选，为数组添加的元素；
2_示例
    let arr = [1, 2, 3, 4];
    let a = arr.splice(1, 2, "hi", "leo");
    console.log(a)      //[ 2, 3 ]
    console.log(arr)    //[ 1, 'hi', 'leo', 4 ]
3_注意:会破坏原数组的结构

10,操作最后一个元素

1_pop(): 删除并返回数组最后一个元素，改变原数组。
2_push(item): 向数组末尾添加一个或多个元素，改变原数组,返回总个数
3_示例:
    let arr = ["pingan", "leo",'44'];
    let a1 = arr.pop();              
    console.log(a1) //44
    console.log(arr) //[ 'pingan', 'leo' ]
    let a2 = arr.push("robin","hi"); 
    console.log(a2) //4
    console.log(arr) //[ 'pingan', 'leo', 'robin', 'hi' ]
4_注意:
    pop操作符,去掉最后一个元素,返回值为去掉的那个元素
    push操作符,增加最后一个元素,返回值为新增后的总元素

11,添加元素

1_shift(): 删除并返回数组第一个元素，改变原数组。
2_unshift(item): 向数组头部添加一个或多个元素，改变原数组，返回新的数组长度。
3_示例
    let arr = ["pingan", "leo"];
    let a1 = arr.shift();               
    console.log(a1)  //pingan
    console.log(arr) //[ 'leo' ]
    let a2 = arr.unshift("robin","hi"); 
    console.log(a2)  //3
    console.log(arr) //[ 'robin', 'hi', 'leo' ]

12,搜索满足要求元素的第一个索引

1_indexOf():返回的是搜索到的第一个元素的索引
2_lastIndexOf()返回的是搜索到的最后一个元素的索引
3_indexOf(ele[,start])和lastIndexOf(ele[,start])
    ele: 需要搜索的元素。
    start: 开始搜索的索引
4_示例
    let arr = ["hh1", "hh2", "hh2", "hh2", "hh3", "hh4"];
    let a1 = arr.indexOf("hh2");      
    let a2 = arr.lastIndexOf("hh2");  
    let a3 = arr.indexOf("hh2",2);    
    console.log(a1) //1
    console.log(a2) // 3
    console.log(a3) // 2

13,搜索满足要求元素的第一个元素和第一个索引(.find和.findIndex)

1_find()方法:用于找出第一个符合条件的数组成员，参数为一个回调函数，所有成员依次执行该回调函数，返回第一个返回值为true的成员，如果没有一个符合则返回undefined
    参数:回调函数接收三个参数，当前值、当前位置和原数组
    示例:
        [1,2,3,4,5].find((value, index, arr) => {
           return  value > 3
        }); //需要写return
        也可以写:
        [1,2,3,4,5].find( a => a < 3 ); //不需要写return
2_findIndex()方法与find()类似，返回第一个符合条件的数组成员的位置，如果都不符合则返回-1

14,测试数组的元素

1_arr.every(callback):测试数组的所有元素是否都通过了指定函数的测试
    1.callback(val, index, arr) : 需要执行的函数，接收三个参数：
        val : 正在处理的当前元素;
        index : 可选，正在处理的当前元素的索引;
        arr : 可选，正在操作的数组    
     2.示例
        let a = [1, "", "aa", 13, 6];
        let res = a.every(function(val, index, arr){
            return typeof val == "number";
        })
        res;// false   
2_arr.some(callback):测试数组中的某些元素是否通过由提供的函数实现的测试
    1.callback(val, index, arr) : 需要执行的函数，接收三个参数：
        val : 正在处理的当前元素;
        index : 可选，正在处理的当前元素的索引;
        arr : 可选，正在操作的数组;
    2.示例
        let a = [1, "", "aa", 13, 6];
        let res = a.some(function(val, index, arr){
            return typeof val == "number";
        })
        res;// true

15.过滤数组的元素

arr.filter(callback):将所有在过滤函数中返回 true 的数组元素放进一个新数组中并返回
1.callback(val, index, arr) : 需要执行的函数，接收三个参数：
    val : 正在处理的当前元素;
    index : 可选，正在处理的当前元素的索引;
    arr : 可选，正在操作的数组;
2.示例
    let a = [1, "", "aa", 13, 6];
    let res = a.filter(function(val, index, arr){
        return typeof val == "number";
    })
    res;//[1, 13, 6]

16.ES6的拓展运算符

1_语法:...[],分割数组的元素,类似于函数的rest参数的逆运算
2_应用
    1.拓展运算符搭配函数的不定参数
        function f (a, ...b){
           console.log(a, b);
        }
        f(...[1,2,3,4]);//1 [ 2, 3, 4 ]

    2.替代apply方法
            示例1:
        // ES6之前
            function f(a, b, c){...};
            var a = [1, 2, 3];
            f.apply(null, a);
        // ES6之后
            function f(a, b, c){...};
            let a = [1, 2, 3];
            f(...a);

        示例2:
         // ES6之前
            Math.max.apply(null, [3,2,6]);
        // ES6之后
            Math.max(...[3,2,6]);

    3.实现数组的深拷贝
        // 通常情况 浅拷贝
            let a1 = [1, 2];
            let a2 = a1; 
            a2[0] = 3;
            console.log(a1,a2); // [3,2] [3,2]
        // 拓展运算符 深拷贝
            let a11 = [1, 2];
            let a22 = [...a11];
            a22[0] = 3;
            console.log(a11,a22); // [1,2] [3,2]

    4.合并数组:合并后的数组是深拷贝
        let a1 = [1,2];
        let a2 = [3];
        let a3 = [4,5];
        // ES5 
        let a4 = a1.concat(a2, a3);
        // ES6
        let a5 = [...a1, ...a2, ...a3];
        console.log(a4) //[ 1, 2, 3, 4, 5 ]
        console.log(a5) //[ 1, 2, 3, 4, 5 ]
        //打印
        a1[0] = 100;
        console.log(a4) //[ 1, 2, 3, 4, 5 ]
        console.log(a5) //[ 1, 2, 3, 4, 5 ]

    5.与解构赋值结合
        let [a, ...b] = [1, 2, 3, 4]; 
        // a => 1  b => [2,3,4]

17.includes():是否包含特定值

第二个参数为起始位置，默认为0
    [1,2,3].includes(2);     // true
    [1,2,3].includes(3,-1);   // true

18.数组一维化

1_flat()
    flat()用于将数组一维化，返回一个新数组，不影响原数组。
    默认一次只一维化一层数组，若需多层，则传入一个整数参数指定层数。
    若要一维化所有层的数组，则传入Infinity作为参数
    [1, 2, [2,3]].flat();        // [1,2,2,3]
    [1,2,[3,[4,[5,6]]]].flat(2); // [1,2,[3,4,5,6]]        
    [1,2,[3,[4,[5,6]]]].flat(3); // [1,2,3,4,5,6]
    [1,2,[3,[4,[5,6]]]].flat('Infinity'); // [1,2,3,4,5,6]
 2_flatMap()
   1.将原数组每个对象先执行一个函数，在对返回值组成的数组执行flat()方法，返回一个新数组，不改变原数组
   2.示例:
   [2, 3, 4].flatMap((x) => [x, x * 2]);  // [2, 4, 3, 6, 4, 8]
